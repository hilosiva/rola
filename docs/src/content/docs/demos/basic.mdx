    ```html title="index.html"
          <div class="item">
            <p>01</p>
          </div>
          <div class="item">

          </div>
          <div class="item">
            <p>03</p>
          </div>
          <div class="item">

          </div>
          <div class="item">

          <div class="item">

          </div>
        </div>
      </div>
    </div>
    ```
  </TabItem>
  <TabItem label="CSS" icon='seti:css'>
    ```css title="style.css"
    .pin-container {
      min-block-size: 300vb;
    }

    .pin {
      --offset: 10vi;

      position: sticky;
      inset-block-start: 0;
      display: block grid;
      align-content: center;
      min-block-size: 100vb;
      padding-inline: var(--offset);
      overflow: hidden;
    }

    .flex {
      --rola-effect-x-start: 0%;
      --rola-effect-x-end: calc(-100% + (100vi - var(--offset) * 2));

      display: block flex;
      column-gap: 2rem;
      align-items: center;
    }

    .item {
      display: block grid;
      place-content: center;
      flex-shrink: 0;
      inline-size: 200px;
      aspect-ratio: 1;
      border: 1px solid;
      font-size: 1.5rem;
      font-weight: 700;
    }
    ```
